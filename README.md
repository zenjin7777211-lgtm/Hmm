-- 3. HÀM GỬI CHỮ LÊN SERVER (Simulate KeyStroke từng chữ + Enter)
local function SubmitWordToServer(word)
    local upperWord = string.upper(word)
    print("Bot đang gõ từ: " + upperWord)
    
    -- Gõ từng chữ cái một vào game
    for i = 1, #upperWord do
        local letter = string.sub(upperWord, i, i)
        Remote:InvokeServer("keyStroke", letter)
        task.wait(0.05) -- Delay nhỏ để server kịp nhận diện, tránh bị filter hoặc lag
    end
    
    -- Sau khi gõ xong thì bấm Enter để xác nhận đáp án
    task.wait(0.1)
    Remote:InvokeServer("tryAnswer")
end

-- 4. HÀM TÌM TỪ HỢP LỆ TỪ KHO TỪ VỰNG
local function FindValidWord(startChar)
    startChar = string.lower(startChar)
    local list = WordBank[startChar]
    
    if not list then return nil end
    
    -- Lặp qua danh sách từ có sẵn để tìm từ chưa từng dùng
    for _, word in ipairs(list) do
        if not UsedWords[word] then
            UsedWords[word] = true -- Đánh dấu từ này đã sử dụng
            return word
        end
    end
    
    return nil -- Hết từ trong kho ở chữ cái này
end
